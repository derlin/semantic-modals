    title: "confirm",

    positive: "yep",
    negative: "cancel",
    cancelable : true

A confirm modal with html content, basic style and semantic icon. 
var options = {
    title   : "Confirm deletion",
    html    : "<p>are you absolutely sure ?</p><p>You cannot undo this action.</p>",
    positive: "proceed",
    negative: "cancel",
    icon    : "warning sign orange",
    basic   : true,
};

{{basicIconOutput | json}}
A basic modal with angular bindings, non cancelable. 
var options = {
    title: "confirm",
    html: "are you sure you want to delete {{inputs.sensor.name}} (id={{inputs.sensor.id}}) ?",
    positive: "yes",
    basic: true,
    inputs     : {
        sensor: {id: 3, name: "sensor-XXXX"}
    },
    cancelable : false
} );

{{htmlInputOutput | json}}
A modal using a custom content. Internally, semanticModal uses the ng-include directive. 
var options = {
    title: "html include",
    htmlInclude: "partial/editModal_content.html",
                    inputs: {name: "some name", description: "a description", hidden: "hidden"},
    positive: "do it!",
    cancelable : false
} );

    cancelable : false
